```cpp
#include<iostream>
class R{  //클래스 선언 = 객체의 틀을 만드는 것 
private:    
    int w,h;   
public:  
    void Set(int a,int b){
        w = a, h = b;  //Set함수로 생성된 객체의 값을 대입
    }
    int Area(){    
        return w*h
    }
};
int main(){
    R r;  //class R의 객체 r을 생성 
    r.Set(3,4);  class R의 객체 r을 3,4
    std::cout << r.Area();  
}
```
