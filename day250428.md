# STL의 활용
## 컨테이너 , 이터레이터, 알고리즘

### 자동정렬
algorithm
```cpp
//algoritm 자동정렬 
#include<iostream>
#include<vector>
#include<list>
#include<map>
#include<algorithm>

using namespace std;

int main() {
	vector<int>  v = { 1,2,5,4,3 };
	sort(v.begin(), v.end());	//algorithm을 포함시켜 사용하는 자동정렬 알고리즘
	for (auto i : v) {
		cout << i<< " "<<endl;
	}
	return 0;
}
```

### 맵
```cpp
//map(맵)
#include<iostream>
#include<vector>
#include<algorithm>
#include<list>
#include<map>
using namespace std;
int main() {
	map <string, int>  fc;	//map의 선언
	fc["apple"] = 3; //map에 요소 추가하기
	fc["orange"] = 5;
	fc["banana"] = 2;
	for (auto i : fc) {
		cout << i.first << "  " << i.second<<endl;	//first는 앞쪽요소 secont는 뒤쪽요소에 접근
	
	}
}
```
### 이토레이터
```cpp
#include<iostream>
#include<vector>
#include<algorithm>
#include<list>
#include<map>
using namespace std;
int main() {
	map <string, int>  fc;	
	fc["apple"] = 3; 
	fc["orange"] = 5;
	fc["banana"] = 2;
	for (auto it = fc.begin(); it != fc.end(); ++it) {
		cout << it->first << "  " << it->second<<endl;	// 이토레이터는 화살표 연산자로 접근 (포인터)
	
	}
}
```
