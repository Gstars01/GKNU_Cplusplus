```cpp
//메모리를 사용한 cpp 프로그래밍 
#include<iostream>
class CRect {
	int w, h;
public:
	CRect(int, int);
	int Area();
};
CRect::CRect(int a, int b) {
	w = a;
	h = b;
}int CRect::Area() {
	return w * h;
}
//컴파일 타임에 메모리가 정해짐 
//CRect r; Date 영역 , global영역
int main() {	
	//컴파일 타임에 메모리가 정해짐 
	//CRect p; local -> Stack영역

	CRect* p;	//동적 메모리 할당 -> Heap영역 /실행시간에 메모리가 정해짐 
	p = new CRect(3, 4);	//포인터 객체 생성	c언어의 malloc 과 동일
	std::cout << p->Area() << std::endl;	//포인터의 객체는 화살표로 접근 
	delete p;	//메모리 할당 해제 c언어의 free와 동일
}
```

```cpp
//c언어 삼각형 클래스 제작
#include<iostream>
using namespace std;
class CRect {
	int w, h;
public:
	CRect(int, int);
	int Area();
};
CRect::CRect(int a, int b) {
	w = a;
	h = b;
}int CRect::Area() {
	return w * h;
}
class CTri {
	int w, h;
public:
	CTri(int, int);
	int Area();
};
CTri::CTri(int a, int b) {
	w = a;
	h = b;
}int CTri::Area() {
	return w * h/2;
}

int main() {	
	CRect* p;
	CTri* t;
	p = new CRect(2, 3);
	t = new CTri(2, 3);
	cout << p->Area() << endl; //또는 <<'\n', "\n"
	cout << t->Area();
	delete p,t;
}
```
