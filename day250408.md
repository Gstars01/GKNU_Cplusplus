```cpp
//smart pointer
#include<iostream>
using namespace std;
class CPnt {
	int x, y;
public:
	CPnt(int x = 0, int y = 0) :x(x), y(y) {}
	void Out() {
		cout << x << "," << y << "\n";
	}
};

// smart pointer 
int main() {
	//unique_ptr<자료형> 포인터명 = make_unique<자료형>();
	unique_ptr<CPnt> p = std::make_unique<CPnt>();
	p->Out();
}
```
