```cpp
#include<iostream>

using namespace std;

//클래스 
class CRect {
	int width, height;
public:
	CRect(int a, int b) {
		width = a;
		height = b;
	}
	CRect(int a) {
		width = a;
		height = a;
	}
	int mul() {
		return width * height;
	}
};


//구조체 
struct CMath {	//구조체 선언 
	int w, h;
};



//함수
int mul(int a, int b) {
	return a * b;
}	
int mul(int a) {		//함수의 오버로딩(중복정의) 
	return a * a;		//매개변수의 타입, 개수, 순서로 구분함 
}



int main() {
	struct CMath m = { 3,4 };	//구조체의 초기화 
	cout << mul(m.h, m.w) << endl;	//구조체의 값 접근 //endl은 줄바꿈 으로 간단히 이해

	CRect r(3, 4);		//class 사용
	cout << r.mul() << endl;	//클래스의 함수 사용 
	CRect r1(5);		//class 속 생성자 오버로딩
	cout << r.mul() << endl;

	return 0;
}

//c++의 특징 
//class
//object(객체)
//Constructor(생성자)
//overloading(함수의 오버로딩)
```
