### raylib

# raylib를 이용한 사각형, 원 그리기 
```cpp
#include "raylib.h"
int main() {
	//		  가로,세로,창 이름
	InitWindow(640, 480, "GKNU");
	// x표를 누르면 창이 닫힘  
	while (!WindowShouldClose())
	{
		//그리기 시작
		BeginDrawing();
		//x좌표 , y좌표 , 너비 , 높이 , 색상 
		DrawRectangle(200, 120, 240,240, BLUE);
		//x좌표 , y좌표 ,  원의 반지름(radius) , 색상 
		DrawCircle(320, 240, 120, GREEN);  
		// 그리기 끝내기
		EndDrawing();
	}
	//창닫기
	CloseWindow();
}
```
